import React, { useEffect, useRef, useState } from "react";
import { motion, AnimatePresence } from "framer-motion"; // Added AnimatePresence import
import { ArrowRight, ArrowUpRight } from "lucide-react";
import { useNavigate } from "react-router-dom";

const Hero = () => {
  const imgRef = useRef(null);
  const [showImage, setShowImage] = useState(false);
  const [mousePosition, setMousePosition] = useState({ x: 0, y: 0 });
  const [isMobile, setIsMobile] = useState(false);

  // Check if mobile device to disable hover effects
  useEffect(() => {
    const checkMobile = () => {
      setIsMobile(window.innerWidth < 768);
    };
    
    checkMobile();
    window.addEventListener("resize", checkMobile);
    return () => window.removeEventListener("resize", checkMobile);
  }, []);

  // Track mouse position for image preview
  useEffect(() => {
    if (isMobile) return;

    const handleMouseMove = (e) => {
      setMousePosition({ x: e.clientX, y: e.clientY });
    };
    
    window.addEventListener("mousemove", handleMouseMove);
    return () => window.removeEventListener("mousemove", handleMouseMove);
  }, [isMobile]);

  const handleResumeDownload = () => {
    window.open("/docs/resume.pdf", "_blank");
  };

  const navigate = useNavigate();
  const handleProjectClick = () => navigate("/projects");

  // Handle mouse enter with mobile check
  const handleMouseEnter = () => {
    if (!isMobile) {
      setShowImage(true);
    }
  };

  return (
    <section className="relative flex items-center justify-center text-solid min-h-[85vh] md:min-h-[90vh] overflow-hidden px-4 sm:px-6 lg:px-8">
      {/* Floating hover image */}
      <AnimatePresence>
        {showImage && !isMobile && (
          <motion.img
            src="/images/suraj.jpg"
            alt="Suraj preview"
            initial={{ opacity: 0, scale: 0.8 }}
            animate={{ opacity: 1, scale: 1 }}
            exit={{ opacity: 0, scale: 0.8 }}
            transition={{ duration: 0.3, ease: "easeOut" }}
            className="fixed w-32 h-32 object-cover cursor-none rounded-full border-2 border-white shadow-2xl pointer-events-none z-40"
            style={{
              left: mousePosition.x + 10,
              top: mousePosition.y + 20,
            }}
          />
        )}
      </AnimatePresence>

      <motion.div
        initial={{ opacity: 0, y: 50 }}
        animate={{ opacity: 1, y: 0 }}
        transition={{ duration: 0.8, ease: "easeOut" }}
        className="flex flex-col items-start text-center md:text-left max-w-6xl w-full"
      >
        <motion.p
          initial={{ opacity: 0, y: 20 }}
          animate={{ opacity: 1, y: 0 }}
          transition={{ delay: 0.2, duration: 0.6 }}
          className="text-lg md:text-xl text-midcolor/80 tracking-wide uppercase mb-4 w-full"
        >
          Full Stack Web Developer
        </motion.p>

        <motion.div
          initial={{ opacity: 0, y: 30 }}
          animate={{ opacity: 1, y: 0 }}
          transition={{ delay: 0.4, duration: 0.6 }}
          className="relative text-4xl sm:text-5xl md:text-6xl lg:text-7xl font-black leading-tight mb-6 select-none w-full"
        >
          <h1 className="break-words">
            I'M{" "}
            <motion.span
              className="text-solid inline-block cursor-none md:cursor-auto"
              onMouseEnter={handleMouseEnter}
              onMouseLeave={() => setShowImage(false)}
              onTouchStart={handleMouseEnter} // For mobile touch
              onTouchEnd={() => setShowImage(false)}
            >
              SURAJ SAVLE
            </motion.span>
          </h1>
        </motion.div>

        <motion.p
          initial={{ opacity: 0, y: 30 }}
          animate={{ opacity: 1, y: 0 }}
          transition={{ delay: 0.6, duration: 0.6 }}
          className="text-xl md:text-2xl text-midcolor/90 leading-relaxed font-light mb-8 hover:text-solid transition-all duration-500 max-w-3xl"
        >
          I create digital experiences that are fast, accessible, and visually
          compelling. Specializing in React, TypeScript, and modern web
          animation.
        </motion.p>

        <motion.div
          className="flex flex-col sm:flex-row items-center gap-4 w-full sm:w-auto"
          initial={{ opacity: 0, y: 40 }}
          animate={{ opacity: 1, y: 0 }}
          transition={{ delay: 0.8, duration: 0.7 }}
        >
          <motion.button
            whileHover={{ scale: 1.05 }}
            whileTap={{ scale: 0.95 }}
            className="group relative w-full sm:w-auto flex items-center justify-center border-2 border-midcolor text-midcolor px-8 py-4 hover:bg-midcolor hover:text-background transition-all duration-500 font-medium text-lg min-w-[200px]"
            onClick={handleProjectClick}
          >
            <span className="flex items-center gap-3">
              View My Work
              <ArrowRight
                size={20}
                className="transition-transform duration-300 group-hover:translate-x-1"
              />
            </span>
          </motion.button>

          <motion.button
            whileHover={{ scale: 1.05 }}
            whileTap={{ scale: 0.95 }}
            className="relative group w-full sm:w-auto flex items-center justify-center border-2 border-midcolor text-midcolor px-8 py-4 transition-all duration-500 font-medium text-lg min-w-[200px] hover:bg-midcolor hover:text-background"
            onClick={handleResumeDownload}
          >
            <span className="flex items-center gap-2">
              Resume
              <ArrowUpRight
                size={20}
                className="transition-transform duration-300 group-hover:translate-x-1 group-hover:translate-y-[-1px]"
              />
            </span>
            <div className="absolute bottom-0 left-0 w-0 h-0.5 bg-midcolor group-hover:w-full transition-all duration-500 group-hover:hidden" />
          </motion.button>
        </motion.div>
      </motion.div>
    </section>
  );
};

export default Hero;
